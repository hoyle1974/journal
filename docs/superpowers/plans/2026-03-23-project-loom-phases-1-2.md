# Project Loom — Phases 1 & 2: Schema + Sequential Pipeline

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Loom graph caching fields to the schema and implement a strictly sequential waterfall pipeline (`ProcessLogSequential`) that guarantees KG and Task state before generating a response node.

**Architecture:** New node types (`object`, `response`) and graph metadata fields (`relevance_score`, `hot_edges`) are added to the schema. A new `ProcessLogSequential` function in `process_entry.go` orchestrates four sequential stages: persist log → refinery → task worker → response worker. Stage failures in 2 & 3 are logged but do not block stage 4. Existing `ProcessEntry` is untouched.

**Tech Stack:** Go 1.22+, Firestore SDK, `internal/infra`, `memory` package, `internal/agent`

```go
package agent

import (
	"context"

	"github.com/jackstrohm/jot/internal/infra"
	"github.com/hoyle1974/memory"
	"strings"
)

// runTaskWorker scans logContent for commitment signals and auto-creates Task nodes
// after graph objects exist so the task can link to them.
// extractedObjectIDs are the UUIDs of object nodes created by the refinery in this run.
//
// STUB: Phase 4 will add full 2-hop graph context linking.
func runTaskWorker(ctx context.Context, app *infra.App, logContent string, extractedObjectIDs []string) error {
	ctx, span := infra.StartSpan(ctx, "loom.task_worker")
	defer span.End()

	parsed, err := RunEvaluatorExtract(ctx, app, logContent)
	if err != nil {
		infra.LoggerFrom(ctx).Warn("loom task worker: evaluator failed", "error", err)
		return err
	}
	if parsed == nil {
		return nil
	}
	if parsed.FutureCommitment < AgencyTaskCommitmentThreshold {
		return nil
	}
	intent := strings.TrimSpace(parsed.CommitmentIntent)
	if len(intent) < MinCommitmentIntentLen {
		return nil
	}
	t := &memory.Task{
		Content:         intent,
		Status:          memory.TaskStatusPending,
		JournalEntryIDs: extractedObjectIDs,
	}
	taskUUID, err := app.Memory.CreateTask(ctx, t)
	if err != nil {
		infra.LoggerFrom(ctx).Warn("loom task worker: create task failed", "error", err)
		return err
	}
	infra.LoggerFrom(ctx).Info("loom task worker: task created", "task_uuid", taskUUID, "intent", intent)
	return nil
}

// runResponseWorker generates a proactive response node using 2-hop RAG context.
// graphExtractFailed signals that Stage 2 (refinery) did not complete — the response
// should note limited context.
//
// STUB: Phase 4 will add full 2-hop retrieval and LLM call.
func runResponseWorker(ctx context.Context, app *infra.App, logUUID, logContent string, graphExtractFailed bool) error {
	ctx, span := infra.StartSpan(ctx, "loom.response_worker")
	defer span.End()

	if graphExtractFailed {
		infra.LoggerFrom(ctx).Info("loom response worker: skipped — graph extraction failed; stub will not generate response", "log_uuid", logUUID)
		// Phase 4: still attempt with degraded context.
		return nil
	}
	// Phase 4 will: vector-search top-5 relationships, expand hot_edges, fetch pending tasks,
	// call LLM, and store a NodeTypeResponse node with logic_trace.
	infra.LoggerFrom(ctx).Info("loom response worker: stub — no response node generated yet", "log_uuid", logUUID)
	return nil
}
```

```go
// ProcessLogSequential runs the Loom waterfall pipeline for a new log entry.
// It guarantees stages complete in strict order before returning.
//
//   Stage 1: Persist the log node to Firestore (immediate write).
//   Stage 2: Refinery — extract KG triples and commit graph objects. (required before Stage 3)
//   Stage 3: Task Worker — scan for commitments and create task nodes linked to graph objects.
//   Stage 4: Response Worker — 2-hop RAG retrieval + response node generation.
//
// Stage 2 and 3 failures are logged and reported in the returned report but do NOT
// abort the pipeline — Stage 4 always runs, noting degraded context when prior stages fail.
func ProcessLogSequential(ctx context.Context, app *infra.App, logUUID, logContent, timestamp, source string) (*ProcessEntryReport, error) {
	ctx, span := infra.StartSpan(ctx, "loom.process_log_sequential")
	defer span.End()
	span.SetAttributes(map[string]string{
		"log_uuid": logUUID,
		"source":   source,
	})

	if app == nil || app.Config() == nil {
		return nil, fmt.Errorf("ProcessLogSequential: app or config is nil")
	}

	infra.LoggerFrom(ctx).Info("loom pipeline start",
		"event", "loom_start",
		"log_uuid", logUUID,
		"source", source,
	)

	// ── Stage 1: Log Persistence ──────────────────────────────────────────────
	// Write the raw log node immediately. Embedding is added by the existing
	// process-entry task path; here we ensure the doc exists before graph work.
	stage1Span := infra.SpanFromContext(ctx)
	_ = stage1Span // span already active on ctx
	infra.LoggerFrom(ctx).Debug("loom stage 1: log persistence", "log_uuid", logUUID)
	client, err := app.Firestore(ctx)
	if err != nil {
		return nil, fmt.Errorf("loom stage 1: firestore client: %w", err)
	}
	logDoc := map[string]any{
		"content":             logContent,
		"source":              source,
		"timestamp":           timestamp,
		"node_type":           "log",
		"significance_weight": 0.3,
	}
	if _, setErr := client.Collection(memory.EntriesCollection).Doc(logUUID).Set(ctx, logDoc, firestore.MergeAll); setErr != nil {
		return nil, fmt.Errorf("loom stage 1: write log node: %w", setErr)
	}
	infra.LoggerFrom(ctx).Info("loom stage 1 done: log node persisted", "log_uuid", logUUID)

	// ── Stage 2: Refinery ─────────────────────────────────────────────────────
	infra.LoggerFrom(ctx).Debug("loom stage 2: refinery", "log_uuid", logUUID)
	refineryErr := runRefineryPipeline(ctx, app, logUUID, logContent)
	if refineryErr != nil {
		infra.LoggerFrom(ctx).Warn("loom stage 2 FAILED: refinery pipeline error — continuing to stage 3",
			"log_uuid", logUUID,
			"error", refineryErr,
		)
	} else {
		infra.LoggerFrom(ctx).Info("loom stage 2 done: refinery complete", "log_uuid", logUUID)
	}

	// ── Stage 3: Task Worker ──────────────────────────────────────────────────
	// Pass the log UUID as the object ID set so tasks can backlink to the log node.
	// Phase 4 will pass actual extracted object IDs from the refinery result.
	infra.LoggerFrom(ctx).Debug("loom stage 3: task worker", "log_uuid", logUUID)
	taskErr := runTaskWorker(ctx, app, logContent, []string{logUUID})
	if taskErr != nil {
		infra.LoggerFrom(ctx).Warn("loom stage 3 FAILED: task worker error — continuing to stage 4",
			"log_uuid", logUUID,
			"error", taskErr,
		)
	} else {
		infra.LoggerFrom(ctx).Info("loom stage 3 done: task worker complete", "log_uuid", logUUID)
	}

	// ── Stage 4: Response Worker ──────────────────────────────────────────────
	graphExtractFailed := refineryErr != nil
	infra.LoggerFrom(ctx).Debug("loom stage 4: response worker", "log_uuid", logUUID, "graph_extract_failed", graphExtractFailed)
	responseErr := runResponseWorker(ctx, app, logUUID, logContent, graphExtractFailed)
	if responseErr != nil {
		infra.LoggerFrom(ctx).Warn("loom stage 4: response worker error",
			"log_uuid", logUUID,
			"error", responseErr,
		)
	} else {
		infra.LoggerFrom(ctx).Info("loom stage 4 done: response worker complete", "log_uuid", logUUID)
	}

	infra.LoggerFrom(ctx).Info("loom pipeline complete",
		"event", "loom_done",
		"log_uuid", logUUID,
		"stage2_ok", refineryErr == nil,
		"stage3_ok", taskErr == nil,
		"stage4_ok", responseErr == nil,
	)

	report := &ProcessEntryReport{
		Content: utils.TruncateString(logContent, 500),
		Source:  source,
	}
	return report, nil
}
```
